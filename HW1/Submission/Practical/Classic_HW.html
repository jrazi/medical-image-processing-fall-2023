def add_salt_and_pepper_noise(image, noise_ratio=0.02):
    noisy_image = image.copy()
    h, w= noisy_image.shape
    noisy_pixels = int(h * w * noise_ratio)

    for _ in range(noisy_pixels):
        row, col = np.random.randint(0, h), np.random.randint(0, w)
        if np.random.rand() < 0.5:
            noisy_image[row, col] = 0
        else:
            noisy_image[row, col] = 255

    return noisy_image

import numpy as np
from scipy import fftpack

def low_pass(I, cutoff):
    # Perform the Fourier transform
    I_fft = fftpack.fftn(I)
    I_fft_shifted = fftpack.fftshift(I_fft)

    # Create a mask for the frequencies we want to keep
    rows, cols = I.shape
    crow, ccol = int(rows/2), int(cols/2)
    mask = np.zeros((rows, cols))
    mask[crow-cutoff:crow+cutoff, ccol-cutoff:ccol+cutoff] = 1

    # Apply the mask to the shifted Fourier transform
    I_fft_masked = I_fft_shifted * mask

    # Shift the frequencies back to original arrangement
    I_fft_masked_unshifted = fftpack.ifftshift(I_fft_masked)

    # Perform the inverse Fourier transform
    I_df = np.real(fftpack.ifftn(I_fft_masked_unshifted))

    return I_df

I_df = low_pass(I, cutoff=30)

Question (5)¶

wich denoising method performed better? how is it related to the type and nature of the noise?

assign your chosen result of denoising to I for next section.

Answer:

The spatial filter (specifically the median filter) has performed significantly better. This is because salt and pepper noise is effectively random occurrences of black and white pixels, and the median filter is particularly well-suited to removing this type of noise. (Outlier data) It does this by replacing each pixel value with the median value of the pixel values in its neighborhood.

The frequency filter (low pass filter), on the other hand, is more effective for reducing high-frequency noise such as Gaussian noise, which is not the case here.

So, based on the nature of the noise, the spatial filter should perform better for denoising in this case.

def high_pass(I, cutoff):
    # Compute the low pass filtered image
    I_low = low_pass(I, cutoff)

    # Subtract the low pass filtered image from the original image
    I_high = I - I_low

    return I_high

from scipy import ndimage

def sharpen(I, alpha=30):
    # Apply a Gaussian blur to the image, to reduce the noise a bit more
    blurred = ndimage.gaussian_filter(I, sigma=0.1)

    # Compute the Laplacian of the blurred image
    laplacian = ndimage.laplace(I)

    # Sharpen the image by subtracting the Laplacian from the original image
    sharpened = I + alpha * laplacian

    # Clip the values of the sharpened image to the range [0, 255]
    sharpened = np.clip(sharpened, 0, 255)

    return sharpened.astype(np.uint8)

I_sharp_spatial = sharpen(I, alpha=5)

from skimage.transform import pyramid_gaussian, resize

def multi_enhance(I, max_layer=5, cutoff=30):
    # Create a Gaussian pyramid
    pyramid = tuple(pyramid_gaussian(I, max_layer=max_layer))

    # Initialize the enhanced image as the original image
    I_enhanced = I.copy().astype(float)

    # Iterate over the layers of the pyramid
    for layer in range(max_layer):
        # Compute the high-pass filtered image for this layer
        I_high = high_pass(pyramid[layer], cutoff=cutoff)

        # Resize the high-pass filtered image to match the original image size
        I_high_resized = np.clip(resize(I_high, I.shape), 0, 255)

        # Add the high-pass filtered image to the enhanced image
        I_enhanced += I_high_resized

    # Clip values outside the range 0-255 and convert back to uint8
    I_enhanced = np.clip(I_enhanced, 0, 255).astype(np.uint8)

    return I_enhanced

I_e = multi_enhance(I, max_layer=5, cutoff=30)
